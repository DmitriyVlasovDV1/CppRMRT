cmake_minimum_required(VERSION 3.25)
project(glut_test)

set(CMAKE_CXX_STANDARD 20)

find_package(OpenGL REQUIRED)
find_package(GLEW REQUIRED)
find_package(glfw3 REQUIRED)

set(SOURCE_FILES src/def.h src/main.cpp)

include_directories(${PROJECT_SOURCE_DIR}/include)
link_directories(${PROJECT_SOURCE_DIR}/lib)

add_executable(glut_test ${SOURCE_FILES})

target_include_directories(glut_test PUBLIC -I/${PROJECT_SOURCE_DIR}/include)
target_link_libraries(glut_test PUBLIC GLEW::glew glfw)
